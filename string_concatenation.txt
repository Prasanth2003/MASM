//program to implement string concatenation

DATA SEGMENT
	STRING DB "Concatenated $"
	STRING2 DB "String$"
DATA ENDS

CODE SEGMENT
	ASSUME CS:CODE, DS:DATA
	START:
	MOV AX,DATA
	MOV DS,AX
	LEA DX,STRING
	MOV AH,09H
	INT 21H
	LEA DX,STRING2
	INT 21H
	MOV AH,4CH
	INT 21H
CODE ENDS
END START
